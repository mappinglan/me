//Step 1: Download the Source Code
wget https://cdn.kernel.org/pub/linux/kernel/v5.x/linux-5.10.42.tar.xz
//Step 2: Extract the Source Code
tar xvf linux-5.10.42.tar.xz
//Step 3: Install Required Packages
sudo apt-get install git fakeroot build-essential ncurses-dev xz-utils libssl-dev bc flex libelf-dev bison
//Step 4: Configure Kernel
cd linux-5.10.42
cp -v /boot/config-$(uname -r) .config
make menuconfig
//Step 5: Build the Kernel
make
sudo make modules_install
sudo make install

//错误：No rule to make target ‘debian/certs/benh@debian.org.cert.pem’, needed by ‘certs/x509_certificate_list’
//解决：vim .config,再把上面那行改成：#CONFIG_SYSTEM_TRUSTED_KEYS=”debian/certs/benh@debian.org.cert.pem”
//错误：BTF: .tmp_vmlinux.btf: pahole (pahole) is not available//解决：sudo apt install dwarves

//Step 6: Update the Bootloader (Optional)
sudo update-initramfs -c -k 5.10.42
sudo update-grub
//Step 7: Reboot and Verify Kernel Version
uname -mrs
